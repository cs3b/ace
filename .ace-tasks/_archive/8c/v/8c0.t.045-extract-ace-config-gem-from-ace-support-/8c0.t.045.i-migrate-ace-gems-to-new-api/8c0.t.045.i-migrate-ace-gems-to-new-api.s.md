## Standard Pattern

```ruby
# lib/ace/my_gem.rb
module Ace
  module MyGem
    def self.config
      @config ||= begin
        gem_root = Gem.loaded_specs["ace-my-gem"]&.gem_dir ||
                   File.expand_path("../..", __dir__)
        Ace::Config.create(
          config_dir: ".ace",
          defaults_dir: ".ace-defaults",
          gem_path: gem_root
        ).resolve_for(["my-gem/config.yml"]).data
      end
    end

    def self.reset_config!
      @config = nil
    end
  end
end
```
