## Technical Approach

### Architecture Pattern

VariableInterpolator follows ATOM Atom pattern:
- Pure function: no side effects, no external dependencies
- Single responsibility: text interpolation
- Testable in isolation

### Variable Syntax

Use Mustache-style `{{variable}}` for clarity:

```yaml
# preset with variables
```

### VariableInterpolator Atom API

```ruby
module Ace::Coworker::Atoms
  class VariableInterpolator
    VARIABLE_PATTERN = /\{\{([a-z_][a-z0-9_]*)\}\}/i

    # Pure function: interpolate variables in text
    # @param text [String] Text with {{variable}} placeholders
    # @param variables [Hash<Symbol,String>] Variable values
    # @return [String] Text with variables replaced
    def self.interpolate(text:, variables:)
      text.gsub(VARIABLE_PATTERN) do |match|
        var_name = Regexp.last_match(1).downcase.to_sym
        variables.fetch(var_name, match) # Keep original if not found
      end
    end

    # Pure function: extract variable names from text
    # @param text [String] Text with {{variable}} placeholders
    # @return [Array<Symbol>] List of variable names
    def self.extract_variables(text:)
      text.scan(VARIABLE_PATTERN).flatten.map { |v| v.downcase.to_sym }.uniq
    end

    # Pure function: collect variables from multiple sources
    # Priority: CLI > ENV > defaults
    # @param cli_vars [Hash] Variables from --var flags
    # @param env_prefix [String] Environment variable prefix (default: "ACE_VAR_")
    # @param defaults [Hash] Default values
    # @return [Hash<Symbol,String>] Merged variables
    def self.collect_variables(cli_vars: {}, env_prefix: "ACE_VAR_", defaults: {})
      result = defaults.transform_keys { |k| k.to_s.downcase.to_sym }

      # Collect from environment
      ENV.each do |key, value|
        if key.start_with?(env_prefix)
          var_name = key.sub(env_prefix, "").downcase.to_sym
          result[var_name] = value
        end
      end

      # CLI vars take priority
      cli_vars.each do |key, value|
        result[key.to_s.downcase.to_sym] = value
      end

      result
    end
  end
end
```

### Standard Variables

| Variable | Source | Description |
|----------|--------|-------------|
| `{{taskref}}` | --var or ACE_VAR_TASKREF | Task reference (e.g., "148") |
| `{{branch}}` | --var or auto (git) | Current git branch |
| `{{date}}` | auto | Current date (YYYY-MM-DD) |
| `{{session}}` | auto (generated) | Session ID |

### Prepare Command Implementation

```ruby
class Prepare < Dry::CLI::Command
  include Ace::Core::CLI::DryCli::Base

  desc "Prepare a new job configuration from preset"

  option :preset, aliases: ["-p"], required: true,
         desc: "Preset name to use"
  option :var, aliases: ["-v"], type: :array, default: [],
         desc: "Variable assignment (key=value, repeatable)"
  option :output, aliases: ["-o"], default: "job.yaml",
         desc: "Output file path"

  def call(preset:, var: [], output: "job.yaml", **options)
    # 1. Find preset file
    preset_path = find_preset(preset)
    raise ConfigNotFoundError, "Preset '#{preset}' not found" unless preset_path

    # 2. Parse CLI variables
    cli_vars = parse_var_args(var)

    # 3. Collect all variables (CLI > ENV > auto)
    auto_vars = {
      date: Date.today.iso8601,
      branch: current_git_branch,
      session: generate_session_id
    }
    variables = Atoms::VariableInterpolator.collect_variables(
      cli_vars: cli_vars,
      defaults: auto_vars
    )

    # 4. Read and interpolate preset
    content = File.read(preset_path)
    interpolated = Atoms::VariableInterpolator.interpolate(
      text: content,
      variables: variables
    )

    # 5. Write output file
    File.write(output, interpolated)

    puts "Created #{output} from preset '#{preset}'"
    puts "Variables: #{variables.keys.join(', ')}"
  end

  private

  def parse_var_args(var_array)
    var_array.each_with_object({}) do |assignment, hash|
      key, value = assignment.split("=", 2)
      hash[key.to_sym] = value if key && value
    end
  end
```

```yaml
# work-on-task.yml
name: work-on-task

variables:

    required: true

  branch:
    required: false

  - name: work-on-task

    instructions:
      - "Work on task {{taskref}}."

  - name: finalize
    skill: ace:commit
    instructions:
      - Commit all changes with a descriptive message.
      - Ensure all tests pass before completing.
```
