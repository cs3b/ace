NAME
  ace-b36ts encode - Encode a timestamp to a compact ID (2-8 characters)

USAGE
  ace-b36ts encode [TIMESTAMP] [OPTIONS]

ARGUMENTS
  [TIMESTAMP]                       Timestamp to encode (ISO, readable, 'now', or empty for current time)

OPTIONS
  --format=VALUE, -f VALUE          Output format: 2sec (default), month, week, day, 40min, 50ms, ms
  --count=VALUE, -n VALUE           Generate N sequential IDs starting from timestamp
  --split=VALUE                     Split levels for hierarchical output (month,week,day,block)
  --[no-]path-only                  Output only the split path
  --[no-]json                       Output split data as JSON (works with --split or --count)
  --year-zero=VALUE, -y VALUE       Base year for encoding (default: 2000)
  --[no-]quiet, -q                  Suppress non-essential output
  --[no-]verbose, -v                Show verbose output
  --[no-]debug, -d                  Show debug output
  --help, -h                        Show this help

EXAMPLES
  $ ace-b36ts encode '2025-01-06 12:30:00'    # Encode readable timestamp
  $ ace-b36ts encode now                       # Encode current time
  $ ace-b36ts encode --format day '2025-01-06' # Encode to day format
  $ ace-b36ts encode --format month now        # Encode to month format
  $ ace-b36ts encode --format 40min now        # Encode to 40min format
  $ ace-b36ts encode --format 50ms now         # Encode to 50ms format
  $ ace-b36ts encode --count 10 --format ms now      # Generate 10 sequential ms-precision IDs
  $ ace-b36ts encode -n 5 --format day now           # Generate 5 consecutive day IDs
  $ ace-b36ts encode --count 3 --format 2sec --json now  # Generate 3 IDs as JSON array
  $ ace-b36ts encode --split month,week now    # Encode to hierarchical split output
  $ ace-b36ts encode --split month,week now --path-only  # Output only the path
  $ ace-b36ts encode --split month,day now --json        # Output JSON split data
  $ ace-b36ts encode --year-zero 2025 '2025-01-06'  # Encode with custom base year
