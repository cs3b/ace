NAME
  ace-b36ts decode - Decode a compact ID (2-8 characters) to a timestamp

USAGE
  ace-b36ts decode COMPACT_ID [OPTIONS]

ARGUMENTS
  COMPACT_ID                        2-8 character compact ID to decode (auto-detects format) (required)

OPTIONS
  --year-zero=VALUE, -y VALUE       Base year for decoding (default: 2000)
  --format=VALUE, -f VALUE          Output format (readable, iso, timestamp)
  --[no-]split                      Force hierarchical split decoding
  --[no-]quiet, -q                  Suppress non-essential output
  --[no-]verbose, -v                Show verbose output
  --[no-]debug, -d                  Show debug output
  --help, -h                        Show this help

EXAMPLES
  $ ace-b36ts decode i5                        # Decode 2-char month ID
  $ ace-b36ts decode i5v                       # Decode 3-char week ID (3rd char 31-35)
  $ ace-b36ts decode i50                       # Decode 3-char day ID (3rd char 0-30)
  $ ace-b36ts decode i50j                      # Decode 4-char 40min ID
  $ ace-b36ts decode i50jj3                    # Decode 6-char 2sec ID
  $ ace-b36ts decode i50jj3z                   # Decode 7-char 50ms ID
  $ ace-b36ts decode i50jj3zz                  # Decode 8-char ms ID
  $ ace-b36ts decode i50jj3 --format iso       # Decode to ISO format
  $ ace-b36ts decode i50jj3 --format timestamp # Decode to YYYYMMDD-HHMMSS
  $ ace-b36ts decode i5/1/5/j/j3               # Decode split path (auto-detect separators)
  $ ace-b36ts decode i515jj3 --split           # Decode split full string
